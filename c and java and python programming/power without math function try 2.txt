#include <stdio.h>
#include <math.h>

double power(double base, double exponent);

int main() {
    double base, exponent, result, real_value, error;
    
    printf("Enter base: ");
    scanf("%lf", &base);

    printf("Enter exponent: ");
    scanf("%lf", &exponent);

    // calculate power using our implementation
    result = power(base, exponent);

    // calculate power using math library function
    real_value = pow(base, exponent);

    // calculate percentage error
    error = fabs((real_value - result) / real_value) * 100.0;

    // print results
    printf("Result: %.12lf\n", result);
    printf("Real value: %.12lf\n", real_value);
    printf("Percentage error: %.12lf%%\n", error);

    return 0;
}

double power(double base, double exponent) {
    if (exponent == 0.0) {
        return 1.0;
    }
    else if (exponent < 0.0) {
        return 1.0 / power(base, -exponent);
    }
    else {
        double result = 1.0;
        double term = 1.0;
        double x = exponent * log(base);
        for (int i = 1; i <= 100; i++) {
            term *= x / i;
            result += term;
        }
        return result;
    }
}
